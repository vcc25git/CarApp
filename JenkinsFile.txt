pipeline {
	agent any
	
	tools{
		maven 'M2_HOME'
		jdk 'JAVA_HOME'
	
	}
	
	stages {
		stage('Checkout'){
			steps {
				git 'https://github.com/vcc25git/CarApp.git'
			}
		}
		
		stage('Build'){
			steps {
				bat 'mvn clean compile'
			}
		}
		
		stage('Code analysis') {
			withSonarQubeEnv('sonarqube') {
			bat 'mvn sonar:sonar'
			}
		}

		stage('Testing'){
			steps {
			 	bat "test"
			 	junit '**/target/surefire-reports/TEST-*.xml'
			 }
		}

		stage('Package') {
			steps {
				bat "mvn package"
			}
		}		
	}
}
